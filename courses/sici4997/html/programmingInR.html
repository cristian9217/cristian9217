How to generate Pie Chart in R?

# Create data for the graph.

totalC11 <- c(length(hybridC11), length(presencialC11), length(distanceC11))

labels <- c("Hybrid","Presencial","Distance")

piepercent<- round(100* totalC11 / sum(totalC11), 1)

# Plot the chart.

pie(totalC11, labels = piepercent, main = "Modality for the semester C11",col = rainbow(length(totalC11)))

legend("topright", labels, cex = 1.2, fill = rainbow(length(totalC11)))
